Task

зәҝзЁӢзҡ„й—®йўҳпјҡзәҝзЁӢ(Thread)жҳҜз”ЁжқҘеҲӣе»әе№¶еҸ‘зҡ„дёҖз§ҚдҪҺзә§еҲ«е·Ҙе…·пјҢе°Өе…¶еңЁдәҺд»ҘдёӢж–№йқўжңүдёҖдәӣйҷҗеҲ¶пјҡ

  • иҷҪз„¶ејҖе§ӢзәҝзЁӢзҡ„ж—¶еҖҷеҸҜд»Ҙж–№дҫҝең°дј е…Ҙж•°жҚ®пјҢдҪҶжҳҜеҪ“JoinпјҲзӯүеҫ…пјүзҡ„ж—¶еҖҷпјҢеҫҲйҡҫд»ҺзәҝзЁӢиҺ·еҫ—иҝ”еӣһеҖјгҖӮ
    • еҸҜиғҪйңҖиҰҒи®ҫзҪ®дёҖдәӣе…ұдә«еӯ—ж®ө
    • еҰӮжһңжҠӣеҮәејӮеёёпјҢжҚ•иҺ·е’Ңдј ж’ӯиҜҘејӮеёёйғҪеҫҲйә»зғҰ
  • ж— жі•е‘ҠиҜүзәҝзЁӢеңЁз»“жқҹж—¶ејҖе§ӢеҒҡеҸҰеӨ–зҡ„е·ҘдҪңпјҢеҝ…йЎ»иҝӣиЎҢJoinж“ҚдҪңпјҢеңЁиҝӣзЁӢдёӯйҳ»еЎһеҪ“еүҚзҡ„иҝӣзЁӢпјҢзӯүеҫ…зәҝзЁӢз»“жқҹгҖӮ
  • еҜ№жүӢеҠЁеҗҢжӯҘзҡ„жӣҙеӨ§дҫқиө–д»ҘеҸҠйҡҸд№ӢиҖҢжқҘзҡ„й—®йўҳ
    Taskзұ»еҸҜд»ҘеҫҲеҘҪең°и§ЈеҶідёҠиҝ°й—®йўҳгҖӮTaskжҳҜдёҖдёӘзӣёеҜ№й«ҳзә§зҡ„жҠҪиұЎпјҢе®ғд»ЈиЎЁдәҶдёҖдёӘе№¶еҸ‘ж“ҚдҪң(concurrent)пјҢиҜҘж“ҚдҪңеҸҜиғҪз”ұThreadж”ҜжҢҒпјҢжҲ–дёҚз”ұThreadж”ҜжҢҒгҖӮе№¶дё”пјҢTaskжҳҜеҸҜз»„еҗҲзҡ„гҖӮ
  • TasksеҸҜд»ҘдҪҝз”ЁзәҝзЁӢжұ жқҘеҮҸе°‘еҗҜеҠЁе»¶иҝҹ
  • дҪҝз”ЁTaskCompletionSourceпјҢTasksеҸҜд»ҘеҲ©з”Ёеӣһи°ғзҡ„ж–№ејҸеңЁзӯүеҫ…I/Oз»‘е®ҡж“ҚдҪңжҳҜе®Ңе…ЁйҒҝе…ҚзәҝзЁӢгҖӮ

ејҖе§ӢдёҖдёӘTask

ејҖе§ӢдёҖдёӘTaskжңҖз®ҖеҚ•зҡ„ж–№жі•е°ұжҳҜдҪҝз”ЁTask.RunиҝҷдёӘйқҷжҖҒж–№жі•гҖӮ(.NET4.5ејҖе§ӢпјҢ.NET4.0зҡ„ж—¶еҖҷжҳҜTask.Factory.StartNewиҝҷдёӘйқҷжҖҒж–№жі•) дҪҝз”Ёж–№жі•пјҡдј е…ҘдёҖдёӘActionе§”жүҳеҚіеҸҜгҖӮ

Taskй»ҳи®ӨдҪҝз”ЁзәҝзЁӢжұ пјҢд№ҹе°ұжҳҜеҗҺеҸ°зәҝзЁӢгҖӮеҪ“дё»зәҝзЁӢз»“жқҹж—¶пјҢеҲӣе»әзҡ„жүҖжңүд»»еҠЎйғҪдјҡз»“жқҹгҖӮ

static void Main(string[] args)
{
    Task.Run(() => Console.WriteLine("Foo"));
    Console.ReadLine(); // еҸҜд»ҘиҫҫеҲ°йҳ»еЎһзәҝзЁӢзҡ„ж•ҲжһңгҖӮ
    // еҰӮжһңзЁӢеәҸиҝҗиЎҢе®ҢдәҶпјҢз”ұдәҺTaskжҳҜеҗҺеҸ°зәҝзЁӢпјҢTaskд№ҹдјҡиў«е…ій—ӯгҖӮ
}

Task.Runиҝ”еӣһдёҖдёӘTaskеҜ№иұЎпјҢеҸҜд»ҘдҪҝз”Ёе®ғжқҘзӣ‘и§Ҷе…¶иҝҮзЁӢгҖӮTask.Runд№ӢеҗҺжІЎжңүи°ғз”ЁStartпјҢеӣ дёәиҜҘж–№жі•еҲӣе»әзҡ„жҳҜвҖңзғӯвҖқд»»еҠЎ(hot task)пјӣеҸҜд»ҘйҖҡиҝҮTaskзҡ„жһ„йҖ еҮҪж•°еҲӣе»әвҖңеҶ·вҖқд»»еҠЎ(cold task)пјҢдҪҶеҫҲе°‘иҝҷж ·еҒҡгҖӮ

Wait

и°ғз”ЁTaskзҡ„Waitж–№жі•дјҡиҝӣиЎҢйҳ»еЎһзӣҙеҲ°ж“ҚдҪңе®ҢжҲҗпјҲзӣёеҪ“дәҺи°ғз”ЁThreadдёҠзҡ„Joinж–№жі•пјү

static void Main(string[] args)
{
    Task task = Task.Run(() =>
    {
        Thread.Sleep(3000);
        Console.WriteLine("Foo");
    }); //еҲӣе»әдёҖдёӘвҖңзғӯд»»еҠЎвҖқ

    Console.WriteLine(task.IsCompleted); //False

    task.Wait(); //йҳ»еЎһиҮіtaskе®ҢжҲҗж“ҚдҪң

    Console.WriteLine(task.IsCompleted); //True

    // False
    // Foo
    // True
}

Waitд№ҹеҸҜд»ҘжҢҮе®ҡдёҖдёӘи¶…ж—¶ж—¶й—ҙе’ҢеҸ–ж¶Ҳд»ӨзүҢжқҘжҸҗеүҚз»“жқҹзӯүеҫ…гҖӮ

й•ҝж—¶й—ҙиҝҗиЎҢзҡ„д»»еҠЎ

  • й»ҳи®Өжғ…еҶөдёӢпјҢCLRеңЁзәҝзЁӢжұ дёӯиҝҗиЎҢTaskпјҢиҝҷйқһеёёйҖӮеҗҲзҹӯж—¶й—ҙиҝҗиЎҢзҡ„Compute-Boundзұ»е·ҘдҪң
  • й’ҲеҜ№й•ҝж—¶й—ҙиҝҗиЎҢзҡ„д»»еҠЎжҲ–иҖ…йҳ»еЎһж“ҚдҪңпјҲдҫӢеҰӮеүҚйқўзҡ„дҫӢеӯҗпјүпјҢеҸҜд»ҘдёҚйҮҮз”ЁзәҝзЁӢжұ 
  • еҰӮжһңеҗҢж—¶иҝҗиЎҢеӨҡдёӘlong-running tasksпјҲе°Өе…¶жҳҜе…¶дёӯжңүеӨ„дәҺйҳ»еЎһжҖҒзҡ„пјүпјҢйӮЈд№ҲжҖ§иғҪе°ҶдјҡеҸ—еҫҲеӨ§еҪұе“ҚпјҢиҝҷж—¶жңүжҜ”TaskCreationOptions.LongRunningжӣҙеҘҪзҡ„еҠһжі•пјҡ
    • еҰӮжһңд»»еҠЎжҳҜIO-BoundпјҢTaskCompletionSourceе’ҢејӮжӯҘеҮҪж•°еҸҜд»Ҙи®©дҪ з”Ёеӣһи°ғд»ЈжӣҝзәҝзЁӢе®һзҺ°е№¶еҸ‘гҖӮ
    • еҰӮжһңд»»еҠЎжҳҜCompute-BoundпјҢз”ҹдә§иҖ…/ж¶Ҳиҙ№иҖ…йҳҹеҲ—е…Ғи®ёдҪ еҜ№д»»еҠЎзҡ„е№¶еҸ‘жҖ§иҝӣиЎҢйҷҗжөҒпјҢйҒҝе…ҚжҠҠе…¶е®ғзәҝзЁӢе’ҢиҝӣзЁӢйҘҝжӯ»гҖӮпјҲе№¶иЎҢзј–зЁӢпјү

Taskзҡ„иҝ”еӣһеҖј

TaskжңүдёҖдёӘжіӣеһӢеӯҗзұ»еҸ«еҒҡTask<TResult>пјҢд»–е…Ғи®ёеҸ‘еҮәдёҖдёӘиҝ”еӣһеҖјгҖӮ

дҪҝз”ЁFunc<TResult>е§”жүҳжҲ–е…је®№зҡ„LambdaиЎЁиҫҫејҸжқҘи°ғз”ЁTask.Runе°ұеҸҜд»Ҙеҫ—еҲ°Task<TResult>пјҢйҡҸеҗҺеҸҜд»ҘйҖҡиҝҮResultеұһжҖ§жқҘиҺ·еҫ—иҝ”еӣһзҡ„з»“жһңгҖӮ

еҰӮжһңиҝҷдёӘtaskиҝҳжІЎжңүе®ҢжҲҗж“ҚдҪңпјҢи®ҝй—®ResultеұһжҖ§дјҡйҳ»еЎһиҜҘзәҝзЁӢзӣҙеҲ°иҜҘtaskе®ҢжҲҗж“ҚдҪңгҖӮ

static void Main(string[] args)
{
    Task<int> task = Task.Run(() =>
    {
        Console.WriteLIne("Foo");
        return 3;
    }); //LambdaиЎЁиҫҫејҸдёҺFunc<TResult>е…је®№

    int result = task.Result; //еҰӮжһңtaskжІЎе®ҢжҲҗпјҢдјҡйҳ»еЎһеҪ“еүҚиҝӣзЁӢпјҢзӣҙеҲ°иҝ”еӣһз»“жһң
    Console.WriteLine(result); //3
}

Task<TResult>еҸҜд»ҘзңӢдҪңжҳҜдёҖз§ҚжүҖи°“еҫ—вҖңи®ёиҜәвҖқпјҢеңЁе®ғйҮҢйқўеҢ…иЈ№зқҖдёҖдёӘResultпјҢеңЁзЁҚеҗҺзҡ„ж—¶еҖҷе°ұдјҡеҸҳеҫ—еҸҜз”ЁгҖӮ

Taskзҡ„ејӮеёё

дёҚеҗҢдәҺThreadпјҢTaskеҸҜд»ҘеҫҲж–№дҫҝең°дј ж’ӯејӮеёёгҖӮеҰӮжһңtaskйҮҢйқўжҠӣеҮәдәҶigeжңӘеӨ„зҗҶзҡ„ејӮеёёпјҲж•…йҡңпјүпјҢйӮЈд№ҲејӮеёёе°ұдјҡйҮҚж–°жҠӣеҮәз»ҷпјҡ

  • и°ғз”ЁдәҶwait()зҡ„ең°ж–№гҖҒ
  • и®ҝй—®дәҶTask<TResult>зҡ„ResultеұһжҖ§зҡ„ең°ж–№

ж— йңҖйҮҚж–°жҠӣеҮәејӮеёёпјҢйҖҡиҝҮTaskзҡ„IsFaultedе’ҢIsCancelledеұһжҖ§д№ҹеҸҜд»ҘжЈҖжөӢеҮәTaskжҳҜеҗҰеҸ‘з”ҹдәҶж•…йҡңпјҡ

  • еҰӮжһңдёӨдёӘеұһжҖ§йғҪиҝ”еӣһfalseпјҢйӮЈд№ҲжІЎжңүй”ҷиҜҜеҸ‘з”ҹгҖӮ
  • еҰӮжһңIsCanceledдҪҚtrueпјҢйӮЈе°ұиҜҙжҳҺдёҖдёӘOperationCanceledExceptionдёәиҜҘTaskжҠӣеҮәдәҶејӮеёё
  • еҰӮжһңIsFaultedдёәtrueпјҢйӮЈе°ұиҜҙжҳҺеҸҰдёҖдёӘзұ»еһӢзҡ„ејӮеёёиў«жҠӣеҮәпјҢиҖҢExceptionеұһжҖ§д№ҹе°ҶжҢҮжҳҺй”ҷиҜҜ

вҖңиҮӘжІ»вҖқзҡ„Task

вҖңиҮӘжІ»вҖқзҡ„TaskеҲ¶вҖңи®ҫзҪ®е®Ңе°ұдёҚз®ЎдәҶвҖқзҡ„TaskгҖӮжҢҮдёҚйҖҡиҝҮи°ғз”ЁWait()ж–№жі•гҖҒResultеұһжҖ§жҲ–continuationиҝӣиЎҢдјҡеҗҲзҡ„д»»еҠЎгҖӮ

й’ҲеҜ№иҮӘжІ»зҡ„TaskпјҢйңҖиҰҒеғҸThreadдёҖж ·пјҢжҳҫејҸең°еӨ„зҗҶејӮеёёпјҢйҒҝе…ҚеҸ‘з”ҹвҖңжӮ„ж— еЈ°жҒҜвҖқзҡ„ж•…йҡңгҖӮ
иҮӘжІ»TaskдёҠжңӘеӨ„зҗҶзҡ„ејӮеёёз§°дёәжңӘи§ӮеҜҹеҲ°зҡ„ејӮеёёгҖӮ

жңӘи§ӮеҜҹеҲ°зҡ„ејӮеёё

еҸҜд»ҘйҖҡиҝҮе…ЁеұҖзҡ„TaskScheduler.UnobservedTaskExceptionжқҘи®ўйҳ…жңӘи§ӮеҜҹеҲ°зҡ„ејӮеёёгҖӮ

дҪҝз”Ёи¶…ж—¶иҝӣиЎҢзӯүеҫ…зҡ„TaskпјҢеҰӮжһңеңЁи¶…ж—¶еҗҺеҸ‘з”ҹж•…йҡңпјҢйӮЈд№Ҳе®ғе°Ҷдјҡдә§з”ҹдёҖдёӘвҖңжңӘи§ӮеҜҹеҲ°зҡ„ејӮеёёвҖқ

еңЁTaskеҸ‘з”ҹж•…йҡңеҗҺпјҢеҰӮжһңи®ҝй—®Taskзҡ„ExceptionеұһжҖ§пјҢйӮЈд№ҲиҜҘејӮеёёе°ұиў«и®ӨдёәжҳҜвҖңе·Іи§ӮеҜҹеҲ°зҡ„ејӮеёёвҖқ

Continuation

еҪ“Taskз»“жқҹзҡ„ж—¶еҖҷпјҢз»§з»ӯеҒҡе…¶е®ғдәӢжғ…гҖӮContinuationйҖҡеёёжҳҜйҖҡиҝҮеӣһи°ғзҡ„ж–№ејҸе®һзҺ°зҡ„пјҢеҪ“ж“ҚдҪңдёҖз»“жқҹпјҢе°ұејҖе§Ӣжү§иЎҢгҖӮ

еңЁTaskдёҠи°ғз”ЁGetAwaiterдјҡиҝ”еӣһдёҖдёӘawaiterеҜ№иұЎгҖӮе®ғзҡ„OnCompletedж–№жі•дјҡе‘ҠиҜүtaskпјҡвҖңеҪ“дҪ з»“жқҹ/еҸ‘з”ҹж•…йҡңж—¶иҰҒжү§иЎҢе§”жүҳвҖқ

еҸҜд»Ҙе°ҶContinuationйҷ„еҠ еҲ°е·Із»Ҹз»“жқҹзҡ„taskдёҠйқўпјҢжӯӨж—¶Continuationе°Ҷдјҡиў«е®үжҺ’з«ӢеҚіжү§иЎҢгҖӮ

Awaiter

awaiterжҳҜеҸҜд»ҘжҡҙйңІдёӢеҲ—дёӨдёӘж–№жі•е’ҢдёҖдёӘеұһжҖ§зҡ„еҜ№иұЎпјҡ

  • OnCompleted
  • GetResult
  • IsCompletedпјҲboolеұһжҖ§пјү
    е…¶дёӯпјҢOnCompletedжҳҜINotifyCompletionзҡ„дёҖйғЁеҲҶ

еҰӮжһңеҸ‘з”ҹж•…йҡңпјҢи°ғз”Ёawaiter.GetResult()зҡ„ж—¶еҖҷпјҢејӮеёёдјҡйҮҚж–°жҠӣеҮәгҖӮ
ж— йңҖи°ғз”ЁGetResultпјҢtaskзҡ„ResultеұһжҖ§д№ҹеҸҜд»ҘзӣҙжҺҘи®ҝй—®гҖӮ

йқһжіӣеһӢTask

йқһжіӣеһӢзҡ„taskпјҢGetResult()ж–№жі•жңүдёҖдёӘvoidиҝ”еӣһеҖјпјҢеҸӘз”ЁжқҘйҮҚж–°жҠӣеҮәејӮеёёгҖӮ

жңӘе®ҢвҖҰ

async
